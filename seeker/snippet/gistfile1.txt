#! /bin/bash

#sum all numbers from 0 to 1000 where a number can be divided by 3 or 5 but not both
sum=0

for ((i=1; i<=1000; i++));
do
  if (( i % 3 == 0 && i % 5 != 0 )) || ((i % 5 == 0 && i % 3 != 0 ));
  then
        sum=$((sum + i))
  fi
done

echo "$sum"
